TC 1 - Login with valid credentials

Steps:

  • Open login page
  • Enter valid email and password
  • Click Login

Expected Result:

User is redirected to dashboard

TC 2 - Login with empty password

Steps:

  • Open login page
  • Enter email only
  • Click Login

Expected Result:

Error “Password is required” is displayed

Autotests
WebdriverIO

describe('Login Page Tests', () => {

  it('should login successfully with valid credentials', async () => {
    await browser.url('https://example.com/login');
    await $('#email').setValue('user@example.com');
    await $('#password').setValue('Password123');
    await $('#loginBtn').click();
    await expect(browser).toHaveUrlContaining('/dashboard');
  });

  it('should keep user logged in when Remember Me is checked', async () => {
    await browser.url('https://example.com/login');
    await $('#email').setValue('user@example.com');
    await $('#password').setValue('Password123');
    await $('#rememberMe').click();
    await $('#loginBtn').click();
    await browser.refresh();
    await expect(browser).toHaveUrlContaining('/dashboard');
  });

  it('should hide password input text', async () => {
    await browser.url('https://example.com/login');
    const type = await $('#password').getAttribute('type');
    expect(type).toBe('password');
  });

  it('should show error for invalid password', async () => {
    await browser.url('https://example.com/login');
    await $('#email').setValue('user@example.com');
    await $('#password').setValue('wrongPass');
    await $('#loginBtn').click();
  });
